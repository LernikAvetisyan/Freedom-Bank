function normalizeAccount(acctRaw) {
  const a = String(acctRaw || "").toLowerCase()
  if (a === "checking" || a === "credit") return a
  if (a === "main") return "checking"
  return "checking"
}

function getAccountRef(uid, acctRaw) {
  const acct = normalizeAccount(acctRaw)
  return db.collection("users").doc(uid).collection("account").doc(acct)
}

async function deleteAllDocs(colRef, batchSize = 500) {
  let deleted = 0
  let last = null

  for (;;) {
    let q = colRef.orderBy("__name__").limit(batchSize)
    if (last) q = q.startAfter(last)

    const snap = await q.get()
    if (snap.empty) break

    const batch = db.batch()
    for (const doc of snap.docs) {
      batch.delete(doc.ref)
      deleted++
    }

    last = snap.docs[snap.docs.length - 1]
    await batch.commit()

    if (snap.size < batchSize) break
  }

  return deleted
}

async function wipeAccount(uid, acct) {
  const accountRef = getAccountRef(uid, acct)

  const deletedTxns = await deleteAllDocs(accountRef.collection("transactions"))
  const deletedTargets = await deleteAllDocs(accountRef.collection("dailyTargets"))
  const deletedHourlyLocks = await deleteAllDocs(accountRef.collection("hourlyLocks"))
  const deletedManualDaily = await deleteAllDocs(accountRef.collection("manualDaily"))
  const deletedFloorTopups = await deleteAllDocs(accountRef.collection("floorTopups"))
  const deletedLocks = await deleteAllDocs(accountRef.collection("locks"))

  const accSnap = await accountRef.get()
  if (accSnap.exists) await accountRef.delete()

  return {
    account: String(acct),
    deletedTxns,
    deletedTargets,
    deletedHourlyLocks,
    deletedManualDaily,
    deletedFloorTopups,
    deletedLocks,
    deletedAccountDoc: true,
  }
}

async function resetUserFlags(uid) {
  const userRef = db.collection("users").doc(uid)
  await userRef.set(
    {
      simActivatedAt: FieldValue.delete(),
      simActivatedAtClient: FieldValue.delete(),
      lastCadenceDay: FieldValue.delete(),
      lastCadenceAt: FieldValue.delete(),
      linkedToBudgetWiseAI: FieldValue.delete(),
      linkedUpdatedAt: FieldValue.delete(),
    },
    { merge: true }
  )
}

exports.adminWipeAll = onRequest({ timeoutSeconds: 540 }, async (req, res) => {
  res.set("Access-Control-Allow-Origin", "*")
  res.set("Access-Control-Allow-Headers", "Authorization, Content-Type")
  if (req.method === "OPTIONS") return res.status(204).end()
  if (req.method !== "POST") return res.status(405).json({ error: "Method not allowed" })

  try {
    const token = String(req.headers.authorization || "").replace(/^Bearer\s+/i, "").trim()
    if (!FB_API_KEY) return res.status(500).json({ error: "Server not configured: missing server API key" })
    if (!token || token !== String(FB_API_KEY)) return res.status(401).json({ error: "Unauthorized" })

    const body = req.body && typeof req.body === "object" ? req.body : {}
    const targetUid = String(body.uid || "").trim() || null
    const maxUsers = Number.isFinite(Number(body.maxUsers)) ? Math.max(1, Math.min(5000, Number(body.maxUsers))) : 1000
    const accounts = Array.isArray(body.accounts) && body.accounts.length ? body.accounts.map(normalizeAccount) : ["checking", "credit"]

    const results = []
    let processed = 0

    const processUser = async (uid) => {
      const perUser = { uid, accounts: [] }
      for (const acct of accounts) perUser.accounts.push(await wipeAccount(uid, acct))
      await resetUserFlags(uid)
      return perUser
    }

    if (targetUid) {
      results.push(await processUser(targetUid))
      processed = 1
    } else {
      let pageToken = undefined
      while (processed < maxUsers) {
        const out = await auth.listUsers(Math.min(1000, maxUsers - processed), pageToken)
        pageToken = out.pageToken

        for (const u of out.users) {
          results.push(await processUser(u.uid))
          processed++
          if (processed >= maxUsers) break
        }

        if (!pageToken) break
      }
    }

    const totals = results.reduce(
      (acc, r) => {
        acc.users += 1
        for (const a of r.accounts) {
          acc.deletedTxns += Number(a.deletedTxns || 0)
          acc.deletedTargets += Number(a.deletedTargets || 0)
          acc.deletedHourlyLocks += Number(a.deletedHourlyLocks || 0)
          acc.deletedManualDaily += Number(a.deletedManualDaily || 0)
          acc.deletedFloorTopups += Number(a.deletedFloorTopups || 0)
          acc.deletedLocks += Number(a.deletedLocks || 0)
        }
        return acc
      },
      { users: 0, deletedTxns: 0, deletedTargets: 0, deletedHourlyLocks: 0, deletedManualDaily: 0, deletedFloorTopups: 0, deletedLocks: 0 }
    )

    return res.json({ ok: true, processedUsers: processed, totals, sample: results.slice(0, 25) })
  } catch (e) {
    console.error("adminWipeAll error", e)
    return res.status(500).json({ error: String(e && e.message ? e.message : e) })
  }
})
